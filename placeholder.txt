# .pre-commit-config.yaml
repos:
  - repo: local
    hooks:
      - id: end-of-file-fixer
        name: Ensure trailing LF
        entry: bash -c 'for f in "$@"; do
                          [ "$(tail -c1 "$f")" = "" ] || printf "\n" >> "$f";
                        done'
        language: system
        types: [text]
